using UnityEngine;
using UnityEngine.InputSystem; // ?좏삎 ?쒖뒪???ㅼ엫?ㅽ럹?댁뒪
using BattlePvp.Stats;

[RequireComponent(typeof(CharacterController))]
[RequireComponent(typeof(Animator))]
public class PlayerManager : MonoBehaviour
{
    [Header("Movement Settings")]
    [SerializeField] private StatManager _statManager;
    [SerializeField] private float moveSpeed = 5.0f; // 湲곕낯 ?대룞 ?띾룄
    [SerializeField] private float gravity = 9.81f; // 以묐젰 媛?
    [SerializeField] private float rotationSpeed = 10.0f; // ?뚯쟾 ?띾룄

    private CharacterController controller;
    private Animator animator;
    private Vector2 inputVector; // ?좏삎 ?쒖뒪?쒖뿉??諛쏆쓣 Vector2 媛?
    private float velocityY;
    private bool canMove = true; // ?대룞 媛???щ? ?뚮옒洹?

    private readonly int speedHash = Animator.StringToHash("Speed");

    private void Awake()
    {
        controller = GetComponent<CharacterController>();
        animator = GetComponent<Animator>();
        if (_statManager == null) _statManager = GetComponentInParent<StatManager>();
    }

    private void OnEnable()
    {
        if (_statManager != null)
        {
            _statManager.StatsChanged += OnStatsChanged;
            UpdateMoveSpeed();
        }
    }

    private void OnDisable()
    {
        if (_statManager != null)
            _statManager.StatsChanged -= OnStatsChanged;
    }

    private void OnStatsChanged(StatContainer _) => UpdateMoveSpeed();

    private void UpdateMoveSpeed()
    {
        if (_statManager == null) return;
        float agi = _statManager.GetFinalTotal(StatKind.AGI);
        moveSpeed = 3.0f + (agi * 0.04f);
    }

    // Input System 硫붿떆吏 ?섏떊 (SendMessage 諛⑹떇 ?먮뒗 Player Input 而댄룷?뚰듃 ?쒖슜)
    // ?먮뵒?곗뿉???ㅼ젙???≪뀡 ?대쫫??"Move"?쇰㈃ ?⑥닔 ?대쫫? "OnMove"媛 ?⑸땲??
    public void OnMove(InputValue value)
    {
        inputVector = value.Get<Vector2>();
    }

    private void Update()
    {
        if (!canMove) return; // ?대룞 ?좉툑 ?곹깭硫??꾨옒 濡쒖쭅 ?ㅽ뻾 ????
        ApplyMovement();
    }

    // Animator??SendMessage("SetMovementLock", true/false)瑜?諛쏅뒗 ?⑥닔
    public void SetMovementLock(bool isLocked)
    {
        canMove = !isLocked;

        // 怨듦꺽 ?쒖옉 ??isLocked == true) 湲곗〈 ?낅젰??珥덇린?뷀빐??誘몃걚?ъ쭚 諛⑹?
        if (isLocked)
        {
            inputVector = Vector2.zero;
        }
    }

    private void ApplyMovement()
    {
        // 1. ?대룞 諛⑺뼢 怨꾩궛 (Vector2瑜?Vector3 ?섑룊硫댁쑝濡?蹂??
        Vector3 moveDirection = new Vector3(inputVector.x, 0, inputVector.y).normalized;

        // 2. 以묐젰 泥섎━
        if (controller.isGrounded)
            velocityY = -0.5f;
        else
            velocityY -= gravity * Time.deltaTime;

        // 3. 理쒖쥌 ?대룞
        Vector3 finalMove = (moveDirection * moveSpeed) + (Vector3.up * velocityY);
        controller.Move(finalMove * Time.deltaTime);

        // 4. ?뚯쟾 泥섎━
        if (inputVector.sqrMagnitude > 0.001f)
        {
            Quaternion targetRotation = Quaternion.LookRotation(moveDirection);
            transform.rotation = Quaternion.Slerp(transform.rotation, targetRotation, rotationSpeed * Time.deltaTime);
        }

        // 5. ?좊땲硫붿씠??(?낅젰???ш린瑜?洹몃?濡??꾨떖)
        animator.SetFloat(speedHash, inputVector.magnitude);
    }
}
